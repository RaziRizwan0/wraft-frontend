# Typography

We use fonts, fontSizes, fontWeights, lineHeights and letterSpacings to customize the theme. You should use the
Text component
with predefined text styles where you can.

## font-size

export const FontSizeComponent = () => {
  function splitRemValue(remString) {
    const match = remString.match(/^(\d*\.?\d+)/);
    return match ? parseFloat(match[1]) : null;
  }
  return (
    <table>
      <tbody>
        {Object.entries(theme.fontSizes).map(([key, value]) => (
          <tr key={key}>
            <td>{key}</td>
            <td>{value}</td>
            <td>{value && theme.toPx(splitRemValue(value))}</td>
          </tr>
        ))}
      </tbody>
    </table>
  );
};

<FontSizeComponent />
